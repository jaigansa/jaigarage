sudo apt-get update && sudo apt-get install nginx libnginx-mod-rtmp vlc
sudo nano /etc/nginx/nginx.conf

rtmp {
        server {
                listen 1935;
                chunk_size 4096;

                application live {
                        live on;
                        record off;
                }
        }
}

sudo nginx -t && sudo systemctl restart nginx
